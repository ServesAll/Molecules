## Floating Button

<Margin style={{minHeight:'150px'}}>
    <FloatingBtn color="#2d3436">
        <Center>
            <H3 color="#ffffff">></H3>
        </Center>
    </FloatingBtn>
</Margin><br />


```code
import {
    H3,
    Center,
    FloatingBtn,
    LoaderElement,
    useThemeContext,
} from '@servesall/atoms';

function RootElement() {
    const {themeContext} = useThemeContext();
    const primaryColor = themeContext.color1;
    const [active] = useState(false);
    const [loading, setLoading] = useState(false);

  return (
        <FloatingBtn
            active={active}
            color={primaryColor}
            onClick={() => setLoading(true)}
            LoaderElement={<LoaderElement />}
        >
          <Center>
             <H3>></H3>
          </Center>
        </FloatingBtn>
  );
}
```
