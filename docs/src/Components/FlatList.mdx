```code
import {
    FlatList,
    H3,
    useThemeContext,
} from '@servesall/atoms';

function RootElement() {
    const {themeContext} = useThemeContext();
    const primaryColor = themeContext.color1;

    const DividerText = () => {
        return <H3>divider 1</H3>;
    }

    const DividerText2 = () => {
        return <H3>divider 2</H3>;
    }

    const itemsArray = [
        {name: 'hi', Divider: [{name: 'divider', value: DividerText}], Divider2: [{name: 'divider', value: DividerText2}]},
        {name: 'from', Divider: [{name: 'divider', value: DividerText}], Divider2: [{name: 'divider', value: DividerText2}]},
        {name: 'normal', Divider: [{name: 'divider', value: DividerText}], Divider2: [{name: 'divider', value: DividerText2}]},
        {name: 'Flat', Divider: [{name: 'divider', value: DividerText}], Divider2: [{name: 'divider', value: DividerText2}]},
        {name: 'List', Divider: [{name: 'divider', value: DividerText}], Divider2: [{name: 'divider', value: DividerText2}]},
    ];

    const renderTitle = ({name}) => {
        return <H3>{name}</H3>;
    }

    return <FlatList.Line lineColor={primaryColor} itemsArray={itemsArray} ListItemChild={renderTitle} />
  );
}
```
